localStorage : Object Has Keys Values  - entries : Keys and Values of an Object

==Syntax==
object.entries(myObj) :    Enter : Object Return Array |
                           It is a Js method |
                           Return an Array Of Elements Each Element 
                           is an Array of Two Element Key and Value

==Example1==let ObjUser = {name : "Ala" , age : 27};
console.log(object.entries(ObjUser)) // [ ["name" , "Ala"] , ["age" , 27] ]
==Example2==
console.log(Object.entries("foo")); //[[0 , "f"] , [1 , "o"] , [2 , "o"]]
==Example3==
console.log(Object.entries(100)); // []
==Example4== Entries : Keys + Values
const obj = { a: 5, b:"Hello", c: 9 };
for (const [key, value] of Object.entries(obj)) {
  console.log(`${key} : ${value}`); // "a : 5", "b : Hello", "c : 9"
}
==Example5== keys / Values
const object1 = {
  a: "somestring",
  b: 42,
  c: false
};

console.log(Object.keys(object1)); //["a" , "b" , "c"]
console.log(Object.values(object1)); // ["somestring", 42, false]
